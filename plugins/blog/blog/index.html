blog

Blog plugin for VuePress, providing article collection, categorization, type filtering, and excerpt generation.

Usage

npm i -D @vuepress/plugin-blog@next
.vuepress/config.ts
import { blogPlugin } from '@vuepress/plugin-blog'

export default {
  plugins: [
    blogPlugin({
      // options
    }),
  ],
}
